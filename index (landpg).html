// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; import "@openzeppelin/contracts/token/ERC721/ERC721.sol"; import "@openzeppelin/contracts/access/Ownable.sol"; /// @title TalentReputation /// @notice Onchain reputation + Badge NFT minting for TalentScout contract TalentReputation is ERC721, Ownable { struct TalentProfile { address wallet; uint256 reputationScore; uint256 jobsCompleted; bool isRegistered; } struct JobRecord { address recruiter; address talent; string jobTitle; bool delivered; // confirmed by talent bool confirmed; // confirmed by recruiter uint256 completedAt; } mapping(address => TalentProfile) public talents; mapping(uint256 => JobRecord) public jobs; mapping(address => bool) public verifiedRecruiters; mapping(address => uint256[]) public talentBadges; uint256 public jobCounter; uint256 public badgeCounter; uint256 public constant SCORE_PER_JOB = 10; event TalentRegistered(address indexed talent); event RecruiterVerified(address indexed recruiter); event JobCreated(uint256 indexed jobId, address recruiter, address talent); event DeliveryConfirmed(uint256 indexed jobId, address talent); event JobCompleted(uint256 indexed jobId, address talent, uint256 newScore); event BadgeMinted(address indexed talent, uint256 tokenId, string tier); constructor() ERC721("TalentScout Badge", "TSBADGE") Ownable(msg.sender) {} modifier onlyVerifiedRecruiter() { require(verifiedRecruiters[msg.sender], "Not a verified recruiter"); _; } function verifyRecruiter(address r) external onlyOwner { verifiedRecruiters[r] = true; emit RecruiterVerified(r); } function registerTalent() external { require(!talents[msg.sender].isRegistered, "Already registered"); talents[msg.sender] = TalentProfile(msg.sender, 0, 0, true); emit TalentRegistered(msg.sender); } function createJob(address talent, string calldata title) external onlyVerifiedRecruiter returns (uint256) { require(talents[talent].isRegistered, "Talent not registered"); uint256 id = jobCounter++; jobs[id] = JobRecord(msg.sender, talent, title, false, false, 0); emit JobCreated(id, msg.sender, talent); return id; } function confirmDelivery(uint256 jobId) external { JobRecord storage j = jobs[jobId]; require(j.talent == msg.sender, "Not your job"); require(!j.delivered, "Already delivered"); j.delivered = true; emit DeliveryConfirmed(jobId, msg.sender); } /// Recruiter confirms → score +10 → badge minted at milestones function confirmCompletion(uint256 jobId) external onlyVerifiedRecruiter { JobRecord storage j = jobs[jobId]; require(j.recruiter == msg.sender, "Not your job"); require(j.delivered, "Talent has not confirmed delivery"); require(!j.confirmed, "Already confirmed"); j.confirmed = true; j.completedAt = block.timestamp; TalentProfile storage t = talents[j.talent]; t.jobsCompleted += 1; t.reputationScore += SCORE_PER_JOB; emit JobCompleted(jobId, j.talent, t.reputationScore); uint256 n = t.jobsCompleted; if (n == 1 || n == 5 || n == 10 || n == 25) _mintBadge(j.talent, n); } function _mintBadge(address talent, uint256 n) internal { string memory tier = n == 1 ? "First Hire" : n == 5 ? "Proven Pro" : n == 10 ? "Elite Talent" : "TS Legend"; uint256 id = badgeCounter++; _mint(talent, id); talentBadges[talent].push(id); emit BadgeMinted(talent, id, tier); } function getProfile(address t) external view returns (TalentProfile memory) { return talents[t]; } }
🔒 TalentEscrow.sol
Trustless hiring payments. Recruiter locks ETH in the contract; funds release automatically on confirmed delivery. Built-in dispute handling — our platform arbitrates any disagreement and releases to the rightful party.
TalentEscrow.sol — Solidity ^0.8.20 — Base Network
// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; /// @title TalentEscrow /// @notice Trustless payment escrow for TalentScout hires on Base contract TalentEscrow { enum State { AWAITING, DELIVERED, COMPLETE, DISPUTED, REFUNDED } struct Escrow { address payable recruiter; address payable talent; uint256 amount; // net amount after fee uint256 fee; // 2.5% to platform State state; uint256 deadline; bool talentDelivered; bool recruiterApproved; } mapping(uint256 => Escrow) public escrows; uint256 public counter; address public platform; uint256 public constant FEE_BPS = 250; // 2.5% uint256 public constant AUTO_RELEASE = 7 days; // auto-pays if recruiter silent uint256 public constant DISPUTE_WIN = 7 days; // window to raise dispute event EscrowCreated(uint256 id, address recruiter, address talent, uint256 amount); event Delivered(uint256 id, address talent); event Released(uint256 id, address talent, uint256 amount); event DisputeRaised(uint256 id, address by); event Resolved(uint256 id, bool talentWon); event Refunded(uint256 id, address recruiter); constructor(address _platform) { platform = _platform; } /// Recruiter deposits ETH to create escrow function createEscrow(address payable talent, uint256 deadlineDays) external payable returns (uint256) { require(msg.value > 0, "Must send ETH"); require(talent != msg.sender, "Cannot hire yourself"); require(deadlineDays <= 365, "Max 365 days"); uint256 fee = (msg.value * FEE_BPS) / 10000; uint256 amount = msg.value - fee; uint256 id = counter++; escrows[id] = Escrow( payable(msg.sender), talent, amount, fee, State.AWAITING, block.timestamp + deadlineDays * 1 days, false, false ); payable(platform).transfer(fee); emit EscrowCreated(id, msg.sender, talent, amount); return id; } /// Talent marks work as delivered function confirmDelivery(uint256 id) external { Escrow storage e = escrows[id]; require(msg.sender == e.talent, "Only talent"); require(e.state == State.AWAITING, "Wrong state"); e.talentDelivered = true; e.state = State.DELIVERED; emit Delivered(id, msg.sender); } /// Recruiter approves → funds instantly released function approve(uint256 id) external { Escrow storage e = escrows[id]; require(msg.sender == e.recruiter, "Only recruiter"); require(e.state == State.DELIVERED, "Not delivered yet"); e.recruiterApproved = true; e.state = State.COMPLETE; e.talent.transfer(e.amount); emit Released(id, e.talent, e.amount); } /// Auto-release if recruiter ignores delivery for 7 days function autoRelease(uint256 id) external { Escrow storage e = escrows[id]; require(e.state == State.DELIVERED, "Not delivered"); require(block.timestamp > e.deadline + AUTO_RELEASE, "Too early"); e.state = State.COMPLETE; e.talent.transfer(e.amount); emit Released(id, e.talent, e.amount); } /// Either party raises dispute within 7-day window function raiseDispute(uint256 id) external { Escrow storage e = escrows[id]; require(msg.sender == e.recruiter || msg.sender == e.talent, "Not a party"); require(e.state == State.AWAITING || e.state == State.DELIVERED, "Cannot dispute"); require(block.timestamp <= e.deadline + DISPUTE_WIN, "Window passed"); e.state = State.DISPUTED; emit DisputeRaised(id, msg.sender); } /// Platform resolves — sends ETH to winner function resolve(uint256 id, bool releaseTalent) external { require(msg.sender == platform, "Only platform"); Escrow storage e = escrows[id]; require(e.state == State.DISPUTED, "Not disputed"); if (releaseTalent) { e.state = State.COMPLETE; e.talent.transfer(e.amount); emit Released(id, e.talent, e.amount); } else { e.state = State.REFUNDED; e.recruiter.transfer(e.amount); emit Refunded(id, e.recruiter); } emit Resolved(id, releaseTalent); } function getEscrow(uint256 id) external view returns (Escrow memory) { return escrows[id]; } }
